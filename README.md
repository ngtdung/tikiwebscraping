Function to start and close driver

# Global driver to use throughout the script
DRIVER = None

# Wrapper to close driver if its created
def close_driver():
    global DRIVER
    if DRIVER is not None:
        DRIVER.close()
    DRIVER = None

# Function to (re)start driver
def start_driver(force_restart=False):
    global DRIVER
    
    if force_restart:
        close_driver()
    
    # Setting up the driver
    options = webdriver.ChromeOptions()
    options.add_argument('-headless') # we don't want a chrome browser opens, so it will run in the background
    options.add_argument('-no-sandbox')
    options.add_argument('-disable-dev-shm-usage')

    DRIVER = webdriver.Chrome('chromedriver',options=options)

def scrape_cat(page_url):
  DRIVER.get(page_url)
  categories = []
  content = DRIVER.find_element(By.CLASS_NAME, 'content')
  slide = content.find_elements(By.TAG_NAME, 'a')
  for element in slide:
    temp = {}
    temp['Name'] = element.find_element(By.TAG_NAME, 'div').get_attribute('innerHTML')
    temp['URL'] = element.get_attribute('href')
    categories.append(temp)
  return categories

# Function to extract product info from the product
def get_product_info_single(product_item):
    """ 
    Extract info from a single product element from the driver.
    Args:
        product_item: (WebDriverElement) the product whose info needs to be 
                      extracted.
    Returns:
        info: (dict) a dictionary of info of the product. Every product
              should at least have four pieces of information: name, price,
              link to the product page, and link to the product image.
    """
    info = {'name':'',
            'price':'',
            'product_url':'',
            'image':'',
            'sales':'',
            'discount':'',
            'tikinow': '',
            'freeship':'',
            'official':'',
            'trusted':'',
            'under_price':'',
            'installments':'',
            'gifts':''}
    info_element = product_item.find_element(By.CLASS_NAME, 'info')
    thumbnail_element = product_item.find_element(By.CLASS_NAME, 'thumbnail')
    thumbnail_img = thumbnail_element.find_elements(By.CLASS_NAME, 'styles__StyledDynamicIconBadge-sc-cxexs7-0')

    try:
    # get name
      info['name'] = product_item.find_element(By.CLASS_NAME, 'name').find_element(By.TAG_NAME, 'h3').get_attribute('innerHTML')

    # get price
      info['price'] = product_item.find_element(By.CLASS_NAME, 'price-discount__price').get_attribute('innerHTML')
    
    # get product link    
      info['product_url'] = product_item.get_attribute('href')

    # get thumbnail url
      info['image'] = thumbnail_element.find_element(By.CLASS_NAME, 'image-wrapper').find_element(By.TAG_NAME, 'img').get_attribute('src')

    # get number of sales
      find = info_element.find_element(By.CLASS_NAME, 'styles__StyledQtySold-sc-732h27-2').get_attribute('innerHTML')
      sale = find[7:]
      info['sales'] = sale

    except NoSuchElementException:
      print('No Element')

    #get discount
    try:
        has_discount = info_element.find_element(By.CLASS_NAME, 'has-discount')
        discount = has_discount.find_element(By.CLASS_NAME, 'price-discount__discount').text
        info['discount'] = discount
    except Exception:
        info['discount'] = 0
    
    #get TIKINOW
    try:
      info['tikinow'] = bool(info_element.find_element(By.CLASS_NAME, 'badge-service'))
    except NoSuchElementException:
      info['tikinow'] = False
    
    
    #get Freeship
    try:
      info['freeship'] = thumbnail_img[0].get_attribute('src') == 'https://salt.tikicdn.com/ts/upload/dc/0d/49/3251737db2de83b74eba8a9ad6d03338.png'
    except Exception:
      info['freeship'] = False  
    
    #get OFFICIAL
    try:
      info['official'] = thumbnail_img[-1].get_attribute('src') == 'https://salt.tikicdn.com/ts/upload/b9/1f/4b/557eac9c67a4466ccebfa74cde854215.png'
    except Exception:
      info['official'] = False  
    
    #get TRUSTED
    try:
      info['trusted'] = thumbnail_img[-1].get_attribute('src') == 'https://salt.tikicdn.com/ts/upload/e0/41/da/bb0fc684a838eff5e264ce0534a148f0.png'
    except Exception:
      info['trusted'] = False  

    #get UNDERPRICE
    badge = info_element.find_element(By.CLASS_NAME, 'badge-under-price')
    try: 
      info['under_price'] = bool(badge.find_element(By.CLASS_NAME, 'item'))
    except NoSuchElementException:
      info['under_price'] = False  
    
    #get FREE-GIFTS
    try:
      gift = info_element.find_element(By.CLASS_NAME, 'freegift-list')
      info['gifts'] = bool(gift)
    except NoSuchElementException:
      info['gifts'] = False
    
    #get INSTALLMENTS
    try:
      info['installments'] = bool(info_element.find_element(By.CLASS_NAME, 'badge-benefits').find_element(By.CLASS_NAME, 'item'))
    except NoSuchElementException:
      info['installments'] = False

    return info
